def mape(y_true, y_pred):
    return (abs((y_true - y_pred) / y_true)).mean() * 100

class GetDummy(BaseEstimator, TransformerMixin):
    def __init__(self) -> None:
        super().__init__()
        self.columns = None

    def fit(self, X, y=None):
        self.columns = pd.get_dummies(X).columns
        return self

    def transform(self, X):
        X_new = pd.get_dummies(X, dtype=np.int64).reindex(
            columns=self.columns, fill_value=0)
        return X_new

    def get_feature_names_out(self, input_features=None):
        return self.columns

def correct_type_fuel(df):
    # type, fuelの欠損値の補完
    df.loc[(~df["cylinders"].isin(['3 cylinders', '4 cylinders', '5 cylinders'])) &
           (df["drive"] == '4wd') &
           (df["type"].isin(['truck', 'pickup'])) &
           (df["manufacturer"] == 'ram'), "fuel"] = "diesel"
    df["fuel"] = df["fuel"].fillna("gas")
    # typeの欠損値処理
    df.loc[(df["fuel"] == "diesel"), "type"] = "truck"
    df["type"] = df["type"].fillna("sedan")
    return df

class CleaningData():
    def __init__(self):
        self.fuel_mode = None
        self.odometer_median = None
        self.p_lower = None
        self.p_upper = None
        self.title_status_mode = None
        self.type_mode = None
        self.state_imputers = None

    def fit(self, df):
        self.fuel_mode = df["fuel"].mode().item()
        self.odometer_median = df.loc[df["odometer"] > 0, "odometer"].median()
        self.p_lower = df["odometer"].quantile(0.01)
        self.p_upper = df["odometer"].quantile(0.999)
        self.title_status_mode = df["title_status"].mode().item()
        self.type_mode = df["type"].mode().item()

        df.loc[df["region"] == "northwest KS", "state"] = "ks"
        df.loc[df["region"] == "ashtabula", "state"] = "oh"
        df.loc[df["region"] == "southern WV", "state"] = "wv"
        df.loc[df["region"] == "manhattan", "state"] = "ks"
        state_imputers = {}
        for region in df["region"].unique().tolist():
            imputer = SimpleImputer(
                strategy="most_frequent").set_output(transform="pandas")
            imputer.fit(df.loc[df["region"] == region, ["state"]])
            state_imputers[region] = imputer
        self.state_imputers = state_imputers

    def transform(self, df):
        # correct_year
        this_year = datetime.datetime.today().year
        df.loc[df["year"] > this_year, "year"] -= 1000

        # manufacturerの表記ゆれ修正
        df["manufacturer"] = df["manufacturer"].str.normalize('NFKC')
        df["manufacturer"] = df["manufacturer"].str.lower()
        df["manufacturer"] = df["manufacturer"].str.translate(
            str.maketrans({'ѕ': "s", "α": "a", "а": "a", "о": "o", "ᴄ": "c"}))

        # title_statusの欠損値処理
        df["title_status"] = df["title_status"].fillna(self.title_status_mode)

        # "size"の表記ゆれ修正
        df["size"] = df["size"].str.translate(
            str.maketrans({"ー": "-", "−": "-", }))

        # stateの欠損値代入
        self.missing_region = []
        for region in df["region"].unique().tolist():
            if len(df.loc[(df["region"] == region) & (df["state"].isna())]) > 0:
                if region in self.state_imputers:
                    imputer = self.state_imputers[region]
                    df.loc[df["region"] == region, "state"] = imputer.transform(
                        df.loc[df["region"] == region, ["state"]])
                else:
                    if region not in self.missing_region:
                        self.missing_region.append(region)

        if len(self.missing_region) > 0:
            for region in self.missing_region:
                imputer = SimpleImputer(
                    strategy="most_frequent").set_output(transform="pandas")
                df.loc[(df["region"] == region), ["state"]] = imputer.fit_transform(
                    df.loc[(df["region"] == region), ["state"]])

        df["condition"] = df["condition"].replace(
            {"new": "5_like new",
             "like new": "5_like new",
             "excellent": "4_excellent",
             "good": "3_good",
             "fair": "2_fair",
             "salvage": "1_salvage"})
        df = correct_type_fuel(df)

        # odometerの外れ値処理、クリッピング
        df.loc[df["odometer"] < -100, "odometer"] *= -1
        df.loc[df["odometer"] == -1, "odometer"] = self.odometer_median
        # df["odometer"] = df["odometer"].clip(upper=self.p_upper)
        df = df[df["odometer"]<=self.p_upper].reset_index(drop=True)
        
        return df

    def fit_transform(self, df):
        self.fit(df)
        df = self.transform(df)
        return df

class EngineeringData():
    def __init__(self):
        self.odometer_type_indicator = None
        self.year_type_indicator = None

    def fit(self, df):
        self.year_type_indicator = df.groupby("type")["year"].median().reset_index().rename(columns={"year": "year_indicator"})
        self.odometer_type_indicator = df.groupby("type")["odometer"].median().reset_index().rename(columns={"odometer": "odometer_indicator"})


    def transform(self, df):
        df["condition"] = df["condition"].replace({"new": "like new"})
        df["type"] = df["type"].replace({"bus": "other"})

        df = df.merge(self.year_type_indicator, how="inner", on="type", right_index=False)
        df["year_type_diff"] = df["year"] - df["year_indicator"]
        df = df.drop('year_indicator', axis=1)
        df = df.merge(self.odometer_type_indicator, how="inner", on="type", right_index=False)
        df["odometer_type_diff"] = df["odometer"] - df["odometer_indicator"]
        df = df.drop('odometer_indicator', axis=1)

        return df

    def fit_transform(self, df):
        self.fit(df)
        df = self.transform(df)
        return df


def create_exitence_rate(df, col_list, normalize_col):
    count_df = df.groupby(col_list).size().reset_index(name="count")
    count_df["count"] = count_df.apply(
        lambda row: row["count"] / (df[normalize_col] == row[normalize_col]).sum(), axis=1)
    count_df = count_df.rename(
        columns={"count": f"{'_'.join(col_list)}_existence_rate"})
    return count_df

def create_two_existence_rate(df):
    df_rarity = df.copy()
    cat_col = [
        ("fuel", "size"),
    ]
    crosses = {}
    for col1, col2 in cat_col:
        col_list = [col1, col2]
        cross = create_exitence_rate(df_rarity, col_list, col1)
        df_cross = df_rarity.merge(
            cross, how="inner", on=col_list, right_index=False)
        df_cross = df_cross[df_cross["price"] <= 30000]
        crosses[(col1, col2)] = cross

    return crosses


def merge_existence_rate(df, crosses):
    cat_col = [
        ("fuel", "size"), 
        ]
    for col1, col2 in cat_col:
        cross = crosses[(col1, col2)]
        df = df.merge(cross, how="left", on=[col1, col2], right_index=False)
    return df


def create_one_hot(df):
    df["ohe_manufacturer_type"] = 0
    df.loc[(df["manufacturer"] == "ram") & (df["type"] == "truck"), "ohe_manufacturer_type"] = 1
    return df

class Preprocessor():
    def __init__(self):
        self.cleaner = CleaningData()
        self.engineering = EngineeringData()
        self.crosses = None

    def preprocessing_train(self, df_train):
        df_train = self.cleaner.fit_transform(df_train)
        df_train = self.engineering.fit_transform(df_train)
        self.crosses = create_two_existence_rate(df_train)
        df_train = merge_existence_rate(df_train, self.crosses)
        df_train = create_one_hot(df_train)
        return df_train

    # テストデータ
    def preprocessing_test(self, df_test):
        df_test = self.cleaner.transform(df_test)
        df_test = self.engineering.transform(df_test)
        df_test = merge_existence_rate(df_test, self.crosses)
        df_test = create_one_hot(df_test)
        df_test = df_test.fillna(0)
        return df_test

def xgbpred(df, df_test, fold=5, random_seed=951):
    
    models = []
    train_scores = []
    val_scores = []
    oof_val = np.zeros(df.shape[0])
    oof_train = np.zeros(df.shape[0])
    test_pred = np.zeros(df_test.shape[0])

    drop_list = ["id",  "log_price", "price", "state", "year", "odometer", "size", "cylinders", "transmission"]
    train_feats = df.drop(drop_list, axis=1)
    train_target = df["log_price"]
    X_test = df_test.drop(drop_list, axis=1, errors="ignore")
    cv = KFold(n_splits=fold, random_state=random_seed, shuffle=True)
    for idx, (train_idx, val_idx) in enumerate(cv.split(train_feats, train_target)):
        
        X_train, y_train = train_feats.loc[train_idx], train_target.loc[train_idx]
        X_val, y_val = train_feats.loc[val_idx], train_target.loc[val_idx]

        cat_col = X_train.select_dtypes(exclude=np.number).columns.to_list()
        encoder = TargetEncoder(cols=cat_col)
        # encoder = GetDummy()
        X_train = encoder.fit_transform(X_train, y_train)
        X_val = encoder.transform(X_val)
        X_test = encoder.transform(X_test)

        xgb_params = {
            "learning_rate": 0.011,
            "n_estimators": 100,
            "max_depth": 6,
            "min_child_weight": 20,
            "subsample": 0.88,
            "colsample_bytree": 0.95,
            "booster": "dart",
            "lambda": 2,
            "alpha": 1.8,
            "num_parallel_tree": 6
        }
        model = XGBRegressor(
            objective="reg:pseudohubererror",
            **xgb_params,
            early_stopping_rounds=10, 
            eval_metric=mean_absolute_error,
            verbosity=0,
            )
        model.fit(X_train, y_train, eval_set=[(X_val, y_val)], verbose=0)
        
        y_log_val_pred = model.predict(X_val)
        y_log_train_pred = model.predict(X_train)
        y_log_test_pred = model.predict(X_test)


        oof_train[train_idx] += np.exp(y_log_train_pred) / (fold-1)
        oof_val[val_idx] += np.exp(y_log_val_pred)
        test_pred += np.exp(y_log_test_pred) / fold
        
        train_score = mape(np.exp(y_train), np.exp(y_log_train_pred))
        val_score = mape(np.exp(y_val), np.exp(y_log_val_pred))
        train_scores.append(train_score)
        val_scores.append(val_score)
        models.append(model)
        print(f"Fold: {idx+1}/{fold} train_score: {train_score}, val_score: {val_score}")

    
    train_oof_score = mape(df["price"], oof_train)
    val_oof_score = mape(df["price"], oof_val)
    display(pd.DataFrame([train_scores, val_scores], index=["train", "valid"]).T)
    print(f"train score: {train_oof_score}", f"val score: {val_oof_score}")
    return models, oof_val, test_pred

def nn_pred(df_train, df_test, fold=5, random_seed=951):


    models = []
    train_scores = []
    val_scores = []
    histories = []
    oof_train = np.zeros(df_train.shape[0])
    oof_val = np.zeros(df_train.shape[0])
    test_pred = np.zeros(df_test.shape[0])

    drop_list = ["id",  "log_price", "price", "year", "odometer", "state", "region"]
    train_feats = df_train.drop(drop_list, axis=1, errors="ignore")
    train_target = df_train["price"]
    X_test = df_test.drop(drop_list, axis=1, errors="ignore")


    cv = KFold(n_splits=fold, shuffle=True, random_state=random_seed)
    for idx, (train_idx, val_idx) in enumerate(cv.split(train_feats, train_target)):
        X_train, y_train = train_feats.loc[train_idx], train_target.loc[train_idx]
        X_val, y_val = train_feats.loc[val_idx], train_target.loc[val_idx]

        num_col = X_train.select_dtypes(include=np.number).columns.to_list()
        cat_col = X_train.select_dtypes(exclude=np.number).columns.to_list()
        ct = ColumnTransformer([("ohe", GetDummy(), cat_col),("scaler", StandardScaler(), num_col),],
                                remainder="passthrough").set_output(transform="pandas")
        X_train = ct.fit_transform(X_train)
        X_val = ct.transform(X_val)
        X_test_ = ct.transform(X_test)

        nn_model = tf.keras.models.Sequential([
            tf.keras.layers.Dense(113, activation=tf.nn.relu),
            tf.keras.layers.Dense(64, activation=tf.nn.relu),
            tf.keras.layers.Dropout(0.3),
            tf.keras.layers.Dense(1, activation=tf.identity),
        ])

        early_stopping = tf.keras.callbacks.EarlyStopping(
            monitor='val_loss', patience=5)
        custom_optimizer = tf.keras.optimizers.Adam(learning_rate=0.005)
        nn_model.compile(optimizer=custom_optimizer,
                    loss="mean_absolute_percentage_error",
                    metrics="mean_absolute_percentage_error")
        history = nn_model.fit(
                    X_train,
                    y_train,
                    epochs=50,
                    validation_data=(X_val, y_val),
                    callbacks=[early_stopping],
                    verbose=0
                    )
        
        train_pred = nn_model.predict(X_train, verbose=0)
        valid_pred = nn_model.predict(X_val, verbose=0)
        pred = nn_model.predict(X_test_, verbose=0)

        oof_train[train_idx] += train_pred.flatten() / 4
        oof_val[val_idx] = valid_pred.flatten()
        test_pred += pred.flatten() / 5

        train_score = mape(df_train.loc[train_idx, "price"], train_pred.flatten())
        val_score = mape(df_train.loc[val_idx, "price"], valid_pred.flatten())
        train_scores.append(train_score)
        val_scores.append(val_score)
        histories.append(history)
        models.append(nn_model)
        print(f"Fold: {idx+1}/{fold} train_score: {train_score}, val_score: {val_score}")
    display(pd.DataFrame([train_scores, val_scores], index=["train", "valid"]).T)
    print(f"train: {mape(df_train['price'], oof_train)}, valid: {mape(df_train['price'], oof_val)}")
    return models, oof_val, test_pred, histories
